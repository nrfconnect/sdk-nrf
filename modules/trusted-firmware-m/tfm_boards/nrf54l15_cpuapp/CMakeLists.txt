  PUBLIC
  ${ZEPHYR_NRF_MODULE_DIR}/subsys/nrf_security/src/drivers/cracen/cracenpsa/include
  )

target_sources(platform_s
  PRIVATE
  ${ZEPHYR_BASE}/soc/nordic/nrf54l/soc.c
  )

  PRIVATE
  ${ZEPHYR_BASE}/modules/cmsis/
